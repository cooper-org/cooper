1"""Implementation of the :py:class:`SimultaneousOptimizer` class.""" 

2 

3from typing import Optional 

4 

5from cooper.optim.constrained_optimizers.constrained_optimizer import ConstrainedOptimizer 

6from cooper.optim.optimizer import RollOut 

7 

8 

9class SimultaneousOptimizer(ConstrainedOptimizer): 

10 r"""Optimizes a :py:class:`~cooper.ConstrainedMinimizationProblem` 

11 by performing simultaneous gradient updates to the primal and dual variables. 

12 

13 According to the choice of primal and dual optimizers, the updates are performed as follows: 

14 

15 .. math:: 

16 \vx_{t+1} &= \texttt{primal_optimizer_update} \left( \vx_{t}, \nabla_{\vx} 

17 \Lag_{\text{primal}}(\vx, \vlambda_t, \vmu_t)|_{\vx=\vx_t} \right) 

18 

19 \vlambda_{t+1} &= \left[ \texttt{dual_optimizer_update} \left( \vlambda_{t}, 

20 \nabla_{\vlambda} \Lag_{\text{dual}}({\vx_{t}}, \vlambda, \vmu_t)|_{\vlambda=\vlambda_t} 

21 \right) \right]_+ 

22 

23 \vmu_{t+1} &= \texttt{dual_optimizer_update} \left( \vmu_{t}, \nabla_{\vmu} 

24 \Lag_{\text{dual}}({\vx_{t}}, \vlambda_t, \vmu)|_{\vmu=\vmu_t} \right) 

25 

26 For instance, when the primal/dual updates are gradient descent/ascent on a 

27 :py:class:`~cooper.formulations.Lagrangian` formulation, the updates are as follows: 

28 

29 .. math:: 

30 \vx_{t+1} &= \vx_t - \eta_{\vx} \left [ \nabla_{\vx} f(\vx_t) + \vlambda_t^\top 

31 \nabla_{\vx} \vg(\vx_t) + \vmu_t^\top \nabla_{\vx} \vh(\vx_t) \right ], 

32 

33 \vlambda_{t+1} &= \left [ \vlambda_t + \eta_{\vlambda} \vg(\vx_t) \right ]_+, 

34 

35 \vmu_{t+1} &= \vmu_t + \eta_{\vmu} \vh(\vx_t), 

36 

37 where :math:`\eta_{\vx}`, :math:`\eta_{\vlambda}`, and :math:`\eta_{\vmu}` are step 

38 sizes. 

39 

40 """ 

41 

42 def roll(self, compute_cmp_state_kwargs: Optional[dict] = None) -> RollOut: 

43 """Evaluates the :py:class:`~cooper.CMPState` and performs a simultaneous 

44 primal-dual update. 

45 

46 Args: 

47 compute_cmp_state_kwargs: Keyword arguments to pass to the 

48 :py:meth:`~cooper.ConstrainedMinimizationProblem.compute_cmp_state` 

49 method. 

50 """ 

51 if compute_cmp_state_kwargs is None: 

52 compute_cmp_state_kwargs = {} 

53 self.zero_grad() 

54 

55 cmp_state = self.cmp.compute_cmp_state(**compute_cmp_state_kwargs) 

56 

57 # TODO (gallego-posada): The current design goes over the constraints twice. We 

58 # could reduce overhead by writing a dedicated compute_lagrangian function for 

59 # the simultaneous updates 

60 primal_lagrangian_store = cmp_state.compute_primal_lagrangian() 

61 dual_lagrangian_store = cmp_state.compute_dual_lagrangian() 

62 

63 # The order of the following operations is not important 

64 # because the primal and dual lagrangians have independent gradients 

65 primal_lagrangian_store.backward() 

66 dual_lagrangian_store.backward() 

67 

68 # The order of the following operations is not important too 

69 # because they update independent primal and dual parameters 

70 self.primal_step() 

71 self.dual_step() 

72 

73 return RollOut(cmp_state.loss, cmp_state, primal_lagrangian_store, dual_lagrangian_store)
